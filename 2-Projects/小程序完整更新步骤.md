## 🔧 Phase 4 特殊配置

### 三级决策系统配置

Phase 4会根据风险自动选择决策等级：

| 决策等级 | 执行方式 | 触发条件 |
|---------|---------|---------|
| **AUTO** | 自动执行 | 单笔≤3万 + AI信心≥75% + 回撤≤5% |
| **SEMI** | 通知确认 | 单笔3-10万 或 AI信心60-75% |
| **MANUAL** | 仅通知 | 单笔>10万 或 AI信心<60% 或 回撤>10% |

### 风险参数调整

```javascript
// 配置风险阈值
wx.cloud.callFunction({
    name: 'autoTradingEngine',
    data: {
        action: 'updateConfig',
        params: {
            maxSingleAmount: 50000,      // 单笔最大金额
            maxDailyTrades: 10,          // 日最大交易次数
            maxDrawdown: 0.10,           // 最大回撤10%
            kellyFraction: 0.5,          // Kelly系数（半Kelly）
            autoTradeEnabled: true       // 是否启用自动交易
        }
    }
});
```

```javascript
// pages/trading-console/trading-console.js
Page({
    data: {
        // Phase 1: 市场状态
        marketState: null,

        // Phase 2: 评分列表
        stockScores: [],

        // Phase 3: AI预测
        aiPredictions: {},

        // Phase 4: 持仓和交易
        portfolio: null,
        tradeLogs: [],

        // 控制状态
        autoTradeEnabled: false,
        circuitBreakerActive: false
    },

    async onLoad() {
        await this.loadMarketState();
        await this.loadPortfolio();
        await this.loadTradeLogs();
    },

    // 加载市场状态（Phase 1）
    async loadMarketState() {
        const res = await wx.cloud.callFunction({
            name: 'marketRegimeDetector',
            data: { action: 'predict' }
        });
        this.setData({ marketState: res.result });
    },

    // 分析股票（Phase 1+2+3）
    async analyzeStock(symbol) {
        wx.showLoading({ title: '分析中...' });

        // 1. 市场状态
        const marketRes = await wx.cloud.callFunction({
            name: 'marketRegimeDetector',
            data: { action: 'predict' }
        });
        const marketState = marketRes.result.currentState;

        // 2. 多因子评分
        const scoreRes = await wx.cloud.callFunction({
            name: 'multiFactorScorer',
            data: {
                action: 'score',
                params: { symbol, marketState }
            }
        });

        // 3. AI预测
        const aiRes = await wx.cloud.callFunction({
            name: 'aiPricePredictor',
            data: {
                action: 'predict',
                params: { symbol, marketRegime: marketState }
            }
        });

        // 4. 交易决策（Phase 4）
        const decisionRes = await wx.cloud.callFunction({
            name: 'autoTradingEngine',
            data: {
                action: 'makeDecision',
                params: {
                    signal: {
                        symbol,
                        marketState,
                        score: scoreRes.result.score,
                        aiPrediction: aiRes.result.prediction
                    }
                }
            }
        });

        wx.hideLoading();
        return decisionRes.result;
    },

    // 执行交易（Phase 4）
    async executeTrade(symbol, decision) {
        if (decision.level === 'MANUAL') {
            wx.showModal({
                title: '需要人工确认',
                content: decision.reason,
                confirmText: '执行',
                cancelText: '取消'
            }).then(async (res) => {
                if (res.confirm) {
                    await this.doTrade(symbol, decision);
                }
            });
        } else {
            // AUTO 或 SEMI（已批准）
            await this.doTrade(symbol, decision);
        }
    },

    async doTrade(symbol, decision) {
        wx.showLoading({ title: '交易中...' });

        const res = await wx.cloud.callFunction({
            name: 'autoTradingEngine',
            data: {
                action: 'executeTrade',
                params: {
                    symbol,
                    decision: decision.decision,
                    amount: decision.position
                }
            }
        });

        wx.hideLoading();

        if (res.result.success) {
            wx.showToast({ title: '交易成功', icon: 'success' });
            await this.loadPortfolio();
            await this.loadTradeLogs();
        } else {
            wx.showToast({ title: '交易失败', icon: 'none' });
        }
    },

    // 加载持仓（Phase 4）
    async loadPortfolio() {
        const res = await wx.cloud.callFunction({
            name: 'autoTradingEngine',
            data: { action: 'getPortfolio' }
        });
        this.setData({ portfolio: res.result });
    },

    // 加载交易日志（Phase 4）
    async loadTradeLogs() {
        const res = await wx.cloud.callFunction({
            name: 'autoTradingEngine',
            data: {
                action: 'getAuditLogs',
                params: { limit: 20 }
            }
        });
        this.setData({ tradeLogs: res.result.logs });
    },

    // 切换自动交易（Phase 4）
    async toggleAutoTrade() {
        const enabled = !this.data.autoTradeEnabled;

        await wx.cloud.callFunction({
            name: 'autoTradingEngine',
            data: {
                action: 'updateConfig',
                params: { autoTradeEnabled: enabled }
            }
        });

        this.setData({ autoTradeEnabled: enabled });
        wx.showToast({
            title: enabled ? '已启用自动交易' : '已禁用自动交易',
            icon: 'success'
        });
    }
});
```

```javascript
async function fullTradingTest() {
    console.log('=== 开始完整交易测试 ===');

    // 1. 获取市场状态（Phase 1）
    const marketResult = await wx.cloud.callFunction({
        name: 'marketRegimeDetector',
        data: { action: 'predict' }
    });
    console.log('1. 市场状态:', marketResult.result);

    // 2. 扫描并评分股票（Phase 2）
    const symbols = ['000001', '000002', '600000'];
    const scores = await Promise.all(symbols.map(async symbol => {
        const res = await wx.cloud.callFunction({
            name: 'multiFactorScorer',
            data: {
                action: 'score',
                params: {
                    symbol,
                    marketState: marketResult.result.currentState
                }
            }
        });
        return res.result.score;
    }));
    console.log('2. 评分结果:', scores);

    // 3. AI预测（Phase 3）
    const predictions = await Promise.all(symbols.map(async symbol => {
        const res = await wx.cloud.callFunction({
            name: 'aiPricePredictor',
            data: {
                action: 'predict',
                params: {
                    symbol,
                    marketRegime: marketResult.result.currentState
                }
            }
        });
        return res.result.prediction;
    }));
    console.log('3. AI预测:', predictions);

    // 4. 综合决策（Phase 4）
    const decisions = await Promise.all(symbols.map(async (symbol, i) => {
        const res = await wx.cloud.callFunction({
            name: 'autoTradingEngine',
            data: {
                action: 'makeDecision',
                params: {
                    signal: {
                        symbol,
                        marketState: marketResult.result.currentState,
                        score: scores[i],
                        aiPrediction: predictions[i]
                    }
                }
            }
        });
        return res.result;
    }));
    console.log('4. 交易决策:', decisions);

    // 5. 执行交易（Phase 4）
    for (const decision of decisions) {
        if (decision.level === 'AUTO') {
            console.log(`自动执行: ${decision.symbol} - ${decision.decision}`);
            await wx.cloud.callFunction({
                name: 'autoTradingEngine',
                data: {
                    action: 'executeTrade',
                    params: {
                        symbol: decision.symbol,
                        decision: decision.decision,
                        amount: decision.position
                    }
                }
            });
        }
    }

    console.log('=== 测试完成 ===');
}
```

## 🐛 常见问题

### Q: Phase 4没有创建数据库集合会怎样？

A: 交易功能会报错。必须先创建 `portfolios`、`trade_logs`、`audit_logs` 等集合。

### Q: 如何禁用自动交易？

A: 有两种方法：
1. 设置配置 `autoTradeEnabled: false`
2. 触发熔断（手动或自动）

### Q: AUTO级别的交易安全吗？

A: AUTO级别有多重保护：
- 单笔≤3万
- AI信心≥75%
- 回撤≤5%
- 5重熔断机制

### Q: 如何查看交易历史？

A: 调用 `getAuditLogs` API：
```javascript
wx.cloud.callFunction({
    name: 'autoTradingEngine',
    data: {
        action: 'getAuditLogs',
        params: { limit: 50 }
    }
});
```
